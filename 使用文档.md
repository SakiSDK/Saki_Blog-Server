### Joi
```ts
// 导入 Joi（如果是 TypeScript 需注意导入方式）
import Joi from 'joi';

// 你的验证规则（即前面定义的 schema）
const schema = Joi.object({
    name: Joi.string().min(2).max(128).required().messages({
        'string.empty': '相册名称不能为空',
        'string.min': '相册名称长度不能小于2个字符',
        'string.max': '相册名称长度不能超过128个字符',
        'any.required': '相册名称为必填项'
    }),
    description: Joi.string().allow(null, '').max(500).messages({
        'string.max': '相册描述不能超过500个字符'
    }),
    creator: Joi.string().default('SakiSDK').messages({
        'string.empty': '创建人不能为空'
    })
});

// 待验证的数据（比如前端提交的表单数据）
const data = {
    name: '旅行', // 符合规则
    description: '这是我的旅行照片合集',
    // creator 未传，会使用默认值
};

// 同步验证
const { error, value } = schema.validate(data);

if (error) {
    // 验证失败：输出错误信息
    console.log('验证失败：', error.details[0].message);
} else {
    // 验证成功：使用处理后的数据（包含默认值）
    console.log('验证成功，处理后的数据：', value);
    // 这里可以继续执行后续逻辑（如存入数据库）
}
```


### Op操作符及示例
#### 比较操作符
1. `Op.eq`: 等于(=)
```ts
    user.findAll({
        where: {
            age: {[Op.eq]: 18}
        }
    })
```
2. `Op.ne`: 不等于(!=)
```ts
    user.findAll({
        where: {
            age: {[Op.ne]: 18}
        }
    })
```
3. `Op.gt`: 大于(>)
4. `Op.gte`: 大于等于(>=)
5. `Op.lt`: 小于(<)
6. `Op.lte`: 小于等于(<=)
7. `Op.between`: 在某个范围之间
8. `Op.notBetween`: 不在某个范围之间
9. `Op.in`: 在某个范围之间
10. `Op.notIn`: 不在某个范围之间
11. `Op.like`: 模糊匹配
12. `Op.notLike`: 不模糊匹配
13. `Op.iLike`: 忽略大小写的模糊匹配
14. `Op.notILike`: 忽略大小写的模糊匹配
15. `Op.overlap`: 覆盖
16. `Op.contains`: 包含
17. `Op.contained`: 被包含
18. `Op.any`: 任意
19. `Op.all`: 全部
20. `Op.col`: 列
21. `Op.placeholder`: 占位符
22. `Op.extract`: 提取
23. `Op.date`: 日期
24. `Op.dateAdd`: 日期加减
25. `Op.dateSub`: 日期减
26. `Op.cast`: 转换
27. `Op.fn`: 函数
28. `Op.literal`: 列
29. `Op.now`: 当前时间
30. `Op.random`: 随机数
31. `Op.substring`: 子串
32. `Op.trim`: 去除空格
33. `Op.upper`: 大写
34. `Op.lower`: 小写
35. `Op.overlay`: 覆盖
36. `Op.position`: 位置
37. `Op.regexp`: 正则
39. `Op.replace`: 替换
